function createEditButton(li, row){ const editBtn = document.createElement('button'); editBtn.textContent = 'Edit'; editBtn.className = 'edit-btn'; editBtn.onclick = ()=>{ // Ask for new amount (text input) let input = prompt('Enter amount (prefix + for addition, default is subtraction):', ''); if(input === null) return; // cancel pressed input = input.trim(); if(input === '') return; // empty input let type2 = '-'; // default is subtraction let numStr = input; // If input starts with +, treat as addition if(input.startsWith('+')){ type2 = '+'; numStr = input.slice(1).trim(); } const newAmount = parseFloat(numStr); if(isNaN(newAmount)){ alert('Invalid number!'); return; } // Push the new amount with proper sign li.amounts.push(type2 + newAmount); updateNoteDisplay(li); // Update the wallet balance const balanceCell = row.querySelector('.balance'); let currentBalance = parseFloat(balanceCell.textContent) || 0; currentBalance += (type2 === '+' ? newAmount : -newAmount); balanceCell.textContent = currentBalance.toFixed(2); calculateTotal(); saveAllWallets(); }; return editBtn; }
